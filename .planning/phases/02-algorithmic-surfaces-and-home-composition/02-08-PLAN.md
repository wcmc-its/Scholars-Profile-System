      now: Date = new Date(),
    ): Promise<TopScholarChipData[] | null> {
      const rows = await prisma.publicationAuthor.findMany({
        where: {
          isConfirmed: true,
          OR: [{ isFirst: true }, { isLast: true }],  // D-13 aggregation filter
          scholar: {
            deletedAt: null,
            status: "active",
            roleCategory: { in: [...TOP_SCHOLARS_ELIGIBLE_ROLES] },  // D-14 narrowed (FT only)
          },
          publication: {
            publicationType: { notIn: ["Letter", "Editorial Article", "Erratum"] },
            publicationScores: { some: {} },
            topicAssignments: { some: { topicRef: { slug: topicSlug } } },
          },
        },
        include: {
          publication: { include: { publicationScores: true } },
          scholar: { select: { cwid: true, slug: true, preferredName: true, primaryTitle: true, roleCategory: true } },
        },
      });

      // Aggregate per scholar (D-13 — SUM over first-or-senior papers in topic with top_scholars curve)
      const byCwid = new Map<string, { scholar: typeof rows[number]["scholar"]; total: number }>();
      for (const r of rows) {
        if (!r.scholar) continue;
        const rankable: RankablePublication = {
          pmid: r.publication.pmid,
          publicationType: r.publication.publicationType,
          reciteraiImpact: r.publication.publicationScores[0]?.score ?? 0,
          dateAddedToEntrez: r.publication.dateAddedToEntrez,
          authorship: { isFirst: r.isFirst, isLast: r.isLast, isPenultimate: r.isPenultimate },
          isConfirmed: r.isConfirmed,
        };
        // D-14 — explicitly use the compressed "top_scholars" curve, NOT "recent_highlights"
        const score = scorePublication(rankable, "top_scholars", true, now);
        const entry = byCwid.get(r.cwid!) ?? { scholar: r.scholar, total: 0 };
        entry.total += score;
        byCwid.set(r.cwid!, entry);
      }

      const sorted = [...byCwid.values()]
        .filter((e) => e.total > 0)
        .sort((a, b) => b.total - a.total);

      if (sorted.length < TOP_SCHOLARS_FLOOR) {
        logSparseHide("topic_top_scholars", sorted.length, TOP_SCHOLARS_FLOOR, { topic: topicSlug });
        return null;
      }

      return sorted.slice(0, TOP_SCHOLARS_TARGET).map((e) => ({
        cwid: e.scholar!.cwid,
        slug: e.scholar!.slug,
        preferredName: e.scholar!.preferredName,
        primaryTitle: e.scholar!.primaryTitle,
        identityImageEndpoint: identityImageEndpoint(e.scholar!.cwid),
      }));
    }

    /** RANKING-02 — publication-centric Recent highlights. */
    export async function getRecentHighlightsForTopic(
      topicSlug: string,
      now: Date = new Date(),
    ): Promise<RecentHighlight[] | null> {
      // Group by publication: pull all authors per pub-in-topic, then score each pub once.
      const pubs = await prisma.publication.findMany({
        where: {
          publicationType: { notIn: ["Letter", "Editorial Article", "Erratum"] },
          publicationScores: { some: {} },
          topicAssignments: { some: { topicRef: { slug: topicSlug } } },
        },
        include: {
          publicationScores: true,
          publicationAuthors: {
            where: { isConfirmed: true, scholar: { deletedAt: null, status: "active" } },
            include: { scholar: { select: { cwid: true, slug: true, preferredName: true } } },
            orderBy: [{ isFirst: "desc" }, { isLast: "desc" }],
          },
        },
      });

      const scored = pubs
        .map((p) => {
          const rankable: RankablePublication = {
            pmid: p.pmid,
            publicationType: p.publicationType,
            reciteraiImpact: p.publicationScores[0]?.score ?? 0,
            dateAddedToEntrez: p.dateAddedToEntrez,
            // For publication-centric scoring, authorship doesn't filter — scholarCentric=false makes authorshipWeight return 1.0 always
            authorship: { isFirst: false, isLast: false, isPenultimate: false },
            isConfirmed: true,
          };
          const score = scorePublication(rankable, "recent_highlights", false, now);
          return { pub: p, score };
        })
        .filter((s) => s.score > 0)
        .sort((a, b) => b.score - a.score);

      if (scored.length < RECENT_HIGHLIGHTS_FLOOR) {
        logSparseHide("topic_recent_highlights", scored.length, RECENT_HIGHLIGHTS_FLOOR, { topic: topicSlug });
        return null;
      }

      return scored.slice(0, RECENT_HIGHLIGHTS_TARGET).map(({ pub }) => ({
        pmid: pub.pmid,
        title: pub.title,
        journal: pub.journal,
        year: pub.year,
        pubmedUrl: pub.pubmedUrl ?? null,
        doi: pub.doi ?? null,
        authors: pub.publicationAuthors.slice(0, 5).map((a) => ({
          cwid: a.cwid ?? null,
          slug: a.scholar?.slug ?? null,
          preferredName: a.scholar?.preferredName ?? a.fullName ?? "—",
        })),
        // NO citationCount field — locked by design spec v1.7.1
      }));
    }
