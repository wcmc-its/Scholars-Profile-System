# CANDIDATE (E) ADDENDUM — D-02 LOCKED

D-02 was locked to **candidate (e)** in `02-SCHEMA-DECISION.md`. The plan body below references the candidate-(a) schema in several places (e.g. `topicAssignments.topicRef.parent.slug`, `topicRef: { parentId: { not: null } }` to find subtopics, `prisma.topic.findMany({ where: { parentId: null } })` to find parents). **All such references are superseded.** Under (e):

- The `topic` table contains exactly 67 rows — ALL parents. There is no `parentId` column. Subtopics are NOT first-class.
- The new `publication_topic` table holds the granular `(pmid, cwid, parent_topic_id)` triples with subtopic data embedded as JSON (`primary_subtopic_id`, `subtopic_ids`, `subtopic_confidences`).
- The existing `topic_assignment` table is unchanged (no new `topicId` / `topicRef` relation). Phase 2 surfaces should NOT query through `topic_assignment`; query `publication_topic` directly.
- `publication_topic.author_position` already encodes first/last/middle/etc. — no need to join `publication_author`.

      });

      const scored = rows
        .map((r) => {
          const rankable: RankablePublication = {
            pmid: r.publication.pmid,
            publicationType: r.publication.publicationType,
            reciteraiImpact: r.publication.publicationScores[0]?.score ?? 0,
            dateAddedToEntrez: r.publication.dateAddedToEntrez,
            authorship: { isFirst: r.isFirst, isLast: r.isLast, isPenultimate: r.isPenultimate },
            isConfirmed: r.isConfirmed,
          };
          const score = scorePublication(rankable, "recent_contributions", true, now);
          // resolve parent topic for dedup
          const parentSlug = r.publication.topicAssignments[0]?.topicRef?.parent?.slug ?? null;
          return { row: r, score, parentSlug };
        })
        .filter((s) => s.score > 0 && s.parentSlug !== null)
        .sort((a, b) => b.score - a.score);

      // Dedup one per parent area (spec line 1143)
      const seenParents = new Set<string>();
      const top: typeof scored = [];
      for (const s of scored) {
        if (seenParents.has(s.parentSlug!)) continue;
        seenParents.add(s.parentSlug!);
        top.push(s);
        if (top.length >= RECENT_CONTRIBUTIONS_TARGET) break;
      }

      if (top.length < RECENT_CONTRIBUTIONS_FLOOR) {
        logSparseHide("home_recent_contributions", top.length, RECENT_CONTRIBUTIONS_FLOOR);
        return null;
      }

      return top.map(({ row, score: _score }) => ({
        cwid: row.scholar!.cwid,
        slug: row.scholar!.slug,
        preferredName: row.scholar!.preferredName,
        primaryTitle: row.scholar!.primaryTitle,
        identityImageEndpoint: identityImageEndpoint(row.scholar!.cwid),
        authorshipRole: row.isLast ? "senior author" : "first author",
        paper: {
          pmid: row.publication.pmid,
          title: row.publication.title,
          journal: row.publication.journal,
          year: row.publication.year,
          pubmedUrl: row.publication.pubmedUrl ?? null,
          doi: row.publication.doi ?? null,
        },
      }));
    }

    export async function getSelectedResearch(now: Date = new Date()): Promise<SubtopicCard[] | null> {
      // HOME-02 — Selected research carousel: top-N subtopics by aggregate publication score
      // using the recent_highlights curve (publication-centric), one card per parent area.
      //
      // Query strategy: pull all subtopic-attributed scored publications from the past
      // ~3 years (recency curve hits 0 at 36+ months), score each in JS using
      // scorePublication(_, "recent_highlights", false /* publication-centric */, now),
      // group by parent_id, dedup so only the top subtopic per parent survives,
      // sort by aggregate score, slice top SELECTED_RESEARCH_TARGET (8).

      const horizonMonths = 36;
      const horizon = new Date(now.getTime() - horizonMonths * 30.4 * 24 * 60 * 60 * 1000);

      const rows = await prisma.publication.findMany({
        where: {
          dateAddedToEntrez: { gte: horizon },
          publicationType: { notIn: ["Letter", "Editorial Article", "Erratum"] },
          publicationScores: { some: {} },
          topicAssignments: {
            some: {
              topicRef: { parentId: { not: null } },  // only subtopics
            },
          },
        },
        include: {
          publicationScores: { take: 1 },
          topicAssignments: {
            where: { topicRef: { parentId: { not: null } } },
            include: { topicRef: { include: { parent: true } } },
            take: 1,
          },
          authors: {
            where: { OR: [{ isFirst: true }, { isLast: true }] },
            include: { scholar: { select: { cwid: true, slug: true, preferredName: true } } },
            take: 2,
          },
        },
        take: 5000,  // bounded; horizon already filters most rows
      });

      // Aggregate scores per subtopic
      const subtopicAgg = new Map<string, {
        parentId: string;
        parentSlug: string;
        parentName: string;
        subtopicId: string;
        subtopicSlug: string;
        subtopicName: string;
        score: number;
        publications: Array<{ pmid: string; title: string; firstWcmAuthor: { cwid: string; slug: string; preferredName: string } | null }>;
      }>();

      for (const pub of rows) {
        const ta = pub.topicAssignments[0];
        if (!ta || !ta.topicRef || !ta.topicRef.parentId || !ta.topicRef.parent) continue;
        const subtopic = ta.topicRef;
        const parent = ta.topicRef.parent;
        const rankable: RankablePublication = {
          pmid: pub.pmid,
          publicationType: pub.publicationType,
          reciteraiImpact: pub.publicationScores[0]?.score ?? 0,
          dateAddedToEntrez: pub.dateAddedToEntrez,
          authorship: { isFirst: false, isLast: false, isPenultimate: false },  // publication-centric
          isConfirmed: true,
        };
        const score = scorePublication(rankable, "recent_highlights", false, now);
        if (score <= 0) continue;

        const existing = subtopicAgg.get(subtopic.id) ?? {
          parentId: parent.id,
          parentSlug: parent.slug,
          parentName: parent.name,
          subtopicId: subtopic.id,
          subtopicSlug: subtopic.slug,
          subtopicName: subtopic.name,
          score: 0,
          publications: [] as Array<{ pmid: string; title: string; firstWcmAuthor: { cwid: string; slug: string; preferredName: string } | null }>,
        };
        existing.score += score;
        if (existing.publications.length < 2) {
          const author = pub.authors[0]?.scholar ?? null;
          existing.publications.push({
            pmid: pub.pmid,
            title: pub.title,
            firstWcmAuthor: author,
          });
        }
        subtopicAgg.set(subtopic.id, existing);
      }

      // Dedup: keep the highest-scoring subtopic per parent
      const sorted = Array.from(subtopicAgg.values()).sort((a, b) => b.score - a.score);
      const seenParents = new Set<string>();
      const top: typeof sorted = [];
      for (const s of sorted) {
        if (seenParents.has(s.parentId)) continue;
        seenParents.add(s.parentId);
        top.push(s);
        if (top.length >= SELECTED_RESEARCH_TARGET) break;
      }

      if (top.length < SELECTED_RESEARCH_FLOOR) {
        logSparseHide("home_selected_research", top.length, SELECTED_RESEARCH_FLOOR);
        return null;
      }

      // Aggregate scholar/publication counts per subtopic from a side query.
      const subtopicIds = top.map((t) => t.subtopicId);
      const counts = await prisma.topicAssignment.groupBy({
        by: ["topicId"],
        where: { topicId: { in: subtopicIds } },
        _count: { cwid: true },
      });
      const scholarCountByTopic = new Map(counts.map((c) => [c.topicId, c._count.cwid]));

      return top.map((t) => ({
        parentTopicSlug: t.parentSlug,
        parentTopicName: t.parentName,
        subtopicSlug: t.subtopicSlug,
        subtopicName: t.subtopicName,
        scholarCount: scholarCountByTopic.get(t.subtopicId) ?? 0,
        publicationCount: t.publications.length,
        publications: t.publications,
      }));
    }

    export async function getBrowseAllResearchAreas(): Promise<ParentTopic[]> {
      // D-12: never hidden. D-03: scholar count = distinct active scholars per parent (no eligibility filter).
      const parents = await prisma.topic.findMany({
        where: { parentId: null },
        select: { slug: true, name: true, scholarCount: true },
        orderBy: { name: "asc" },
      });
      // scholarCount is pre-computed by the ETL (Plan 05 stores it on Topic.scholarCount).
      // If pre-computation not yet wired, do an inline aggregation:
      // const counts = await prisma.scholar.groupBy({ ... where: { deletedAt: null, status: "active" } ... });
      return parents.map((p) => ({ slug: p.slug, name: p.name, scholarCount: p.scholarCount }));
    }
